# --- PS 版本检测与兼容函数 ---
$PSVer = $PSVersionTable.PSVersion.Major

function Http-Get($url) {
    if ($PSVer -ge 3) {
        return Invoke-RestMethod -Uri $url -Method Get -TimeoutSec 5
    }
    $wc = New-Object System.Net.WebClient
    try { $wc.Encoding = [System.Text.Encoding]::UTF8 } catch {}
    $wc.DownloadString($url)
}

function Http-PostJson($url, $json) {
    if ($PSVer -ge 3) {
        return Invoke-RestMethod -Uri $url -Method Post -Body $json -ContentType "application/json; charset=utf-8" -TimeoutSec 10
    }
    $wc = New-Object System.Net.WebClient
    try { $wc.Encoding = [System.Text.Encoding]::UTF8 } catch {}
    $wc.Headers.Add("Content-Type", "application/json; charset=utf-8")
    $resp = $wc.UploadString($url, "POST", $json)
    # PS 2.0 UploadString 返回 string, 尝试解析 JSON
    try {
        Add-Type -AssemblyName System.Web.Extensions
        $ser = New-Object System.Web.Script.Serialization.JavaScriptSerializer
        $ser.DeserializeObject($resp)
    } catch {
        $resp
    }
}

function Json-Escape($s) {
    if ($s -eq $null) { return "" }
    $s = "$s"
    $s = $s -replace '\\', '\\'
    $s = $s -replace '"', '\"'
    $s = $s -replace [char]10, '\n'
    $s = $s -replace [char]13, '\r'
    $s = $s -replace [char]9, '\t'
    return $s
}

function Build-Json($obj) {
    if ($PSVer -ge 3) {
        return $obj | ConvertTo-Json -Compress
    }
    # PS 2.0 手动拼接 JSON
    $parts = @()
    foreach ($key in $obj.Keys) {
        $val = Json-Escape $obj[$key]
        $parts += ('"' + $key + '":"' + $val + '"')
    }
    return "{" + ($parts -join ",") + "}"
}

function Get-CimOrWmi {
    param([string]$ClassName)
    if ($PSVer -ge 3) {
        return Get-CimInstance -ClassName $ClassName
    }
    return Get-WmiObject -Class $ClassName
}


function Read-ValueWithDefault {
    param([string]$Default = "")
    if ($Default -ne "") {
        Write-Host "  回车确认: " -NoNewline -ForegroundColor Green
        $buf = New-Object System.Text.StringBuilder -ArgumentList $Default
        [Console]::Write($Default)
        while ($true) {
            $k = [Console]::ReadKey($true)
            if ($k.Key -eq [ConsoleKey]::Enter) {
                [Console]::WriteLine()
                break
            }
            elseif ($k.Key -eq [ConsoleKey]::Escape) {
                [Console]::WriteLine()
                return $Default
            }
            elseif ($k.Key -eq [ConsoleKey]::Backspace -and $buf.Length -gt 0) {
                $buf.Remove($buf.Length - 1, 1) | Out-Null
                [Console]::Write([char]8 + " " + [char]8)
            }
            elseif (-not [char]::IsControl($k.KeyChar)) {
                $buf.Append($k.KeyChar) | Out-Null
                [Console]::Write($k.KeyChar)
            }
        }
        $val = $buf.ToString()
        if ([string]::IsNullOrWhiteSpace($val)) { return $Default }
        return $val
    } else {
        $val = Read-Host "  请输入"
    }
    if ([string]::IsNullOrWhiteSpace($val)) { return $Default }
    return $val
}

function Select-Menu {
    param([string[]]$Items)
    for ($i = 0; $i -lt $Items.Count; $i++) {
        Write-Host "  [$($i+1)] $($Items[$i])"
    }
    while ($true) {
        $sel = Read-Host "  请输入序号 (1-$($Items.Count))"
        $num = 0
        if ([int]::TryParse($sel, [ref]$num) -and $num -ge 1 -and $num -le $Items.Count) {
            return $num - 1
        }
        Write-Host "  [-] 无效输入, 请输入 1-$($Items.Count) 的数字" -ForegroundColor Red
    }
}

# --- 采集硬件信息 ---
Write-Host ""

Write-Host "========================================" -ForegroundColor Cyan

$dm = ""
try {
    $cs = Get-CimOrWmi Win32_ComputerSystem | Select-Object -First 1
    $csp = Get-CimOrWmi Win32_ComputerSystemProduct | Select-Object -First 1
    $dmVer = ""; $dmName = ""; $dmModel = ""; $dmFamily = ""; $dmManu = ""
    if ($csp) { $dmVer = "$($csp.Version)".Trim(); $dmName = "$($csp.Name)".Trim() }
    if ($cs) { $dmModel = "$($cs.Model)".Trim(); $dmFamily = "$($cs.SystemFamily)".Trim(); $dmManu = "$($cs.Manufacturer)".Trim() }
    $oemModel = ""; $biosModel = ""
    try { $v = (Get-ItemProperty 'HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\OEMInformation' -EA 0).Model; if ($v) { $oemModel = "$v".Trim() } } catch {}
    try { $v = (Get-ItemProperty 'HKLM:\HARDWARE\DESCRIPTION\System\BIOS' -EA 0).SystemProductName; if ($v) { $biosModel = "$v".Trim() } } catch {}
    $cands = @($dmVer, $dmName, $dmModel, $dmFamily, "$dmManu $dmFamily", "$dmManu $dmModel", $oemModel, $biosModel)
    foreach ($c in $cands) {
        $t = "$c".Trim()
        if ($t -and $t -notmatch "to be filled|default string|system product name|not applicable") {
            $dm = $t; break
        }
    }
} catch {}

$cpu = ""
try { $cpu = (Get-CimOrWmi Win32_Processor | Select-Object -First 1).Name.Trim() } catch {}

$ramParts = @()
try {
    $mems = Get-CimOrWmi Win32_PhysicalMemory
    foreach ($m in $mems) {
        if ($m.Capacity -gt 0) {
            $gb = [math]::Round($m.Capacity / 1GB)
            $type = "Unknown"
            switch ($m.SMBIOSMemoryType) {
                26 { $type = "DDR4" }
                24 { $type = "DDR3" }
                21 { $type = "DDR2" }
                29 { $type = "DDR5" }
            }
            $spd = ""
            if ($m.Speed -gt 0) { $spd = "-$($m.Speed)MHz" }
            $ramParts += "${gb}GB $type$spd"
        }
    }
} catch {}
$ram = $ramParts -join "; "

$diskParts = @()
try {
    $disks = Get-CimOrWmi Win32_DiskDrive
    foreach ($d in $disks) {
        $mdl = $d.Model.Trim()
        if ($mdl -and $mdl -notmatch "virtual|vhd|vhdx") {
            $gb = [math]::Round($d.Size / 1GB)
            $diskParts += "$mdl ${gb}GB"
        }
    }
} catch {}
$disks = $diskParts -join "; "

$gpuParts = @()
try {
    $gpus = Get-CimOrWmi Win32_VideoController
    foreach ($g in $gpus) {
        $n = $g.Name.Trim()
        if ($n -and $n -notmatch "intel|microsoft|basic display|basic render|radeon\(tm\) graphics$") {
            $gpuParts += $n
        }
    }
} catch {}
$gpu = $gpuParts -join "; "

$macEth = ""
$macWifi = ""
try {
    $adapters = Get-CimOrWmi Win32_NetworkAdapter
    foreach ($a in $adapters) {
        if ($a.MACAddress -and $a.Name -match "以太网|Ethernet|LAN") { $macEth = $a.MACAddress }
        if ($a.MACAddress -and $a.Name -match "Wi-Fi|WLAN|Wireless|无线") { $macWifi = $a.MACAddress }
    }
} catch {}

$hostname = $env:COMPUTERNAME
$osVer = ""
try {
    $os = Get-CimOrWmi Win32_OperatingSystem
    $osVer = "$($os.Caption.Trim()) $($os.OSArchitecture.Trim())"
} catch {}

$swParts = @()
$swSeen = @{}
try {
    $regPaths = @(
        "HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall",
        "HKLM:\SOFTWARE\WOW6432Node\Microsoft\Windows\CurrentVersion\Uninstall",
        "HKCU:\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall"
    )
    foreach ($rp in $regPaths) {
        if (-not (Test-Path $rp)) { continue }
        Get-ChildItem $rp -EA 0 | ForEach-Object {
            $props = Get-ItemProperty $_.PSPath -EA 0
            $dn = ""
            try { $dn = $props.DisplayName.Trim() } catch {}
            if (-not $dn) { return }
            if ($dn -match '^\$\{\{') { return }
            $dv = ""
            try { $dv = $props.DisplayVersion.Trim() } catch {}
            $entry = $dn
            if ($dv) { $entry = "$dn ($dv)" }
            if (-not $swSeen[$entry]) { $swSeen[$entry] = $true; $swParts += $entry }
        }
    }
} catch {}
$swList = ($swParts | Sort-Object) -join "; "

$hw = [PSCustomObject]@{
    device_model       = $dm
    serial_number      = $sn
    cpu                = $cpu
    ram_info           = $ram
    disks_info         = $disks
    gpu                = $gpu
    mac_ethernet       = $macEth
    mac_wifi           = $macWifi
    hostname           = $hostname
    os_version         = $osVer
    installed_software = $swList
}
